The main goal of the project is to create trees in a random but real way. There is the possibility to convert the painting on the canvas to a PNG image file, this is in order to train a neural network whose purpose is to identify regular patterns in plants. The Generator The generator I built knows how to produce patterns while using recursive functions, in this version the initial values ​​are drawn and then rely on patterns of nature



The Main draw algorithm:

function Genrator() {
GEN.ctx.clearRect(0, 0, GEN.canvas.width, GEN.canvas.height);
let center = GEN.canvas.width / 2;
let len = 0;
if (window.innerWidth < 600) {
len = Math.floor(Math.random() * 20) + 120;
GEN.curve = Math.random() * 7 + 3;
} else {
len = Math.floor(Math.random() * 40) + 180;
GEN.curve = Math.random() * 20 + 5;
} GEN.curve2 = Math.random() * 45 + 1;
let angle = 0;
let branchWidth = Math.random() * 55 + 10;
if (Math.random() > 0.5) {
GEN.ctx.globalCompositeOperation = "destination-over";
} else {
GEN.ctx.globalCompositeOperation = "source-over";
}
let color1 =
GEN.color1_array[Math.floor(Math.random() * GEN.color1_array.length)];
let color2 =
GEN.color2_array[Math.floor(Math.random() * GEN.color2_array.length)];
GEN.generateButton.style.background = color1;
GEN.leafDense = Math.floor(Math.random() * 30) + 15;
drawTree( center, GEN.canvas.height - 80, len, angle, branchWidth, color1, color2 );
}
